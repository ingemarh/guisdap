% Transfer from TeX to PS world for different systems:
\iftrue% UNIX
\def\GUPPiCspecials{%
	\special{ps::[begin] gsave \specialscales \addtoPS}%
	\special{ps: plotfile \GUPPiCmacrofile }%
	\special{ps::\parsfromTeXtoPS }%
	\special{ps: plotfile \GUPPiCdatafile }%
	\special{ps::[end] grestore}%
}\fi
\iffalse% Macintosh OzTeX
\def\GUPPiCspecials{%
	\special{\GUPPiCmacrofile firstfile}%
	% firstfile keyword tells OzTeX to not to revert PS world
	\special{\GUPPiCdatafile lastfile \addtoPS \parsfromTeXtoPS initialize }% 
}\fi
\iffalse% Macintosh Textures
\def\GUPPiCspecials{
	\special{postscript \specialscales \addtoPS}%
	\special{illustration \GUPPiCmacrofile }%
	\special{postscript \parsfromTeXtoPS }%
	\special{illustration \GUPPiCdatafile }%
}\fi
